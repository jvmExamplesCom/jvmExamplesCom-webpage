Create Zip File From Directory using ZipOutputStream Example

In this example we will learn how to create a zip file from a directory using ZipOutputStream in Java.

Step 1: Get a List of Files to be Zipped

In order to create a zip file, first we need a list of files to be zipped. In our example, we are going to use File.listFiles() to get a list of all the files in a directory. We will also use an ArrayList for storing these files.


          
           
File dir = new File("C:\\test");
ArrayList<File> fileList = new ArrayList<File>();
getAllFiles(dir, fileList);

Step 2: Create ZipOutputStream

We will now create a ZipOutputStream object by passing FileOutputStream as a parameter.


          
           
FileOutputStream fos = new FileOutputStream("C:\\test\\test.zip");
ZipOutputStream zos = new ZipOutputStream(fos);

Step 3: Iterate through the File List and Add to Zip Entry

In this step, we will iterate through our list of files and add them to the zip entry using ZipOutputStream.putNextEntry() method.


          
           
for (File file : fileList) {
    if (!file.isDirectory()) {
        // create byte buffer
        byte[] buffer = new byte[1024];

        FileInputStream fis = new FileInputStream(file);

        // begin writing a new ZIP entry, positions the stream to the start of the entry data
        zos.putNextEntry(new ZipEntry(file.getName()));

        int length;

        while ((length = fis.read(buffer)) > 0) {
            zos.write(buffer, 0, length);
        }

        zos.closeEntry();

        // close the InputStream
        fis.close();
    }
}

// close the ZipOutputStream
zos.close();

Complete Code


          
           
import java.io.*;
import java.util.ArrayList;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;


public class ZipExample {

    public static void main(String[] args) throws Exception {

        File dir = new File("C:\\test");
        ArrayList<File> fileList = new ArrayList<File>();
        getAllFiles(dir, fileList);

        FileOutputStream fos = new FileOutputStream("C:\\test\\test.zip");
        ZipOutputStream zos = new ZipOutputStream(fos);

        for (File file : fileList) {
            if (!file.isDirectory()) {
                // create byte buffer
                byte[] buffer = new byte[1024];

                FileInputStream fis = new FileInputStream(file);

                // begin writing a new ZIP entry, positions the stream to the start of the entry data
                zos.putNextEntry(new ZipEntry(file.getName()));

                int length;

                while ((length = fis.read(buffer)) > 0) {
                    zos.write(buffer, 0, length);
                }

                zos.closeEntry();

                // close the InputStream
                fis.close();

            }
        }
        // close the ZipOutputStream
        zos.close();
    }

    private static void getAllFiles(File dir, ArrayList<File> fileList) {
        try {
            File[] files = dir.listFiles();
            for (File file : files) {
                fileList.add(file);
                if (file.isDirectory()) {
                    System.out.println("directory:" + file.getCanonicalPath());
                    getAllFiles(file, fileList);
                } else {
                    System.out.println("     file:" + file.getCanonicalPath());
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

}
