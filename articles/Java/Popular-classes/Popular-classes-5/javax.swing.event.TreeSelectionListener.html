javax.swing.event.TreeSelectionListener

javax.swing.event.TreeSelectionListener is an interface in the Java Swing library used to detect when the selection in a tree of nodes changes. By implementing this interface, developers can create a class that contains an event handling method which is notified whenever the selection within a JTree component changes. public class TreeSelectionExample {
  public static void main(String[] args) {
    // Create a TreeModel with two nodes
    DefaultMutableTreeNode rootNode = new DefaultMutableTreeNode("root");
    DefaultMutableTreeNode childNode1 = new DefaultMutableTreeNode("child1");
    DefaultMutableTreeNode childNode2 = new DefaultMutableTreeNode("child2");
    rootNode.add(childNode1);
    rootNode.add(childNode2);
    TreeModel model = new DefaultTreeModel(rootNode);

    // Create a JTree with the TreeModel
    JTree tree = new JTree(model);

    // Register a TreeSelectionListener to be notified when the selection changes
    tree.addTreeSelectionListener(new MyTreeSelectionListener());
  }
}

class MyTreeSelectionListener implements TreeSelectionListener {
  @Override
  public void valueChanged(TreeSelectionEvent e) {
    TreeModel model = e.getTreeModel();
    TreePath path = e.getNewLeadSelectionPath();
    TreePath oldPath = e.getOldLeadSelectionPath();

    System.out.println("New selection path: " + path);
    System.out.println("Old selection path: " + oldPath);
  }
}
