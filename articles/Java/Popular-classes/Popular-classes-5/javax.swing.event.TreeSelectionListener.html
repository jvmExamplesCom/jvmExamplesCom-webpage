javax.swing.event.TreeSelectionListener is an interface in the Java Swing library used to detect when the selection in a tree of nodes changes. public class TreeSelectionExample {
  public static void main(String[] args) {
    // Create a TreeModel with two nodes
    DefaultMutableTreeNode rootNode = new DefaultMutableTreeNode("root");
    DefaultMutableTreeNode childNode1 = new DefaultMutableTreeNode("child1");

    rootNode.add(childNode1);
    rootNode.add(childNode2);
    TreeModel model = new DefaultTreeModel(rootNode);

    // Create a JTree with the TreeModel
    JTree tree = new JTree(model);

    // Register a TreeSelectionListener to be notified when the selection changes
    tree.addTreeSelectionListener(new MyTreeSelectionListener());
  }
}

class MyTreeSelectionListener implements TreeSelectionListener {
  @Override
  public void valueChanged(TreeSelectionEvent e) {

    System.out.println("Old selection path: " + oldPath);
  }
}
