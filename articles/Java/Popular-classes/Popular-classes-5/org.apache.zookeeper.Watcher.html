org.apache.zookeeper.Watcher

process(WatchedEvent event)

onConnected(Watcher.Event.KeeperState connectedState)
    — This method is triggered when a connection is established between the client and the server
  • 
            
             
onDisconnected(Watcher.Event.KeeperState disconnectedState)
