com.squareup.okhttp.MediaType

MediaType textHtmlUtf8 = MediaType.parse("text/html; charset=utf-8");

MediaType applicationJson = MediaType.create("application", "json");

boolean isTextHtml = textHtmlUtf8.match("text/html");
