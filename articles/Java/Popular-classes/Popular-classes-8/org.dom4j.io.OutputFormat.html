Using OutputFormat

String indent = "  ";
boolean omitDeclaration = false;
boolean omitEncoding = false;
String lineSeparator = System.getProperty("line.separator");
boolean trimText = false;

Element root = new Element("root");
doc.add(root);

// Create OutputFormat
