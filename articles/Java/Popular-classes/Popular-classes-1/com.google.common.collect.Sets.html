List<String> list = Lists.newArrayList("hello", "world", "foo");
Set<String> set = Sets.newHashSet(list);

Combining Sets

union()

Set<String> set1 = Sets.newHashSet("a", "b");

Set<String> result = Sets.union(set1, set2);
