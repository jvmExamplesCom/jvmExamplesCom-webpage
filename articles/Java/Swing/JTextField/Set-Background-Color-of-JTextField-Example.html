MyFrame Class

import javax.swing.*;
public class MyFrame extends JFrame {
    public MyFrame() {
        setSize(400, 300);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(null);
        setVisible(true);
    }
}

setBackground()

import java.awt.Color;
import javax.swing.*;
public class BackgroundExample extends MyFrame {
    public BackgroundExample() {
        JTextField txtField = new JTextField();
        txtField.setBounds(50, 50, 100, 30);
        txtField.setBackground(Color.GREEN);
        add(txtField);
    }

        new BackgroundExample();
    }
